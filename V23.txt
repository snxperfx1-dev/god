// FU / FLIP detector — dominant rejection wick at a local extreme (swept OR not).
f_fuPool(simple float _wf, simple int _lb) =>
    var float _tip = na
    var float _bH = na
    var float _bL = na
    var float _mid = na
    var int   _dir = 0
    var bool  _have = false
    var bool  _conf = false
    float _rng = math.max(high - low, 1e-10)
    float _pHi = ta.highest(high, _lb)[1]
    float _pLo = ta.lowest(low,  _lb)[1]
    float _uw  = (high - math.max(open, close)) / _rng
    float _lw  = (math.min(open, close) - low)  / _rng
    bool  _localTop = high >= ta.highest(high, _lb)
    bool  _localBot = low  <= ta.lowest(low,  _lb)
    bool  _bear = _uw >= _wf and ((not na(_pHi) and high >= _pHi and close < _pHi) or (_localTop and close < open))
    bool  _bull = _lw >= _wf and ((not na(_pLo) and low  <= _pLo and close > _pLo) or (_localBot and close > open))
    if _bear
        _dir := -1
        _tip := high
        _bH := math.max(open, close)
        _bL := math.min(open, close)
        _mid := _bH + (_tip - _bH) * 0.5
        _have := true
        _conf := false
    else if _bull
        _dir := 1
        _tip := low
        _bH := math.max(open, close)
        _bL := math.min(open, close)
        _mid := _tip + (_bL - _tip) * 0.5
        _have := true
        _conf := false
    if _have and _dir == -1 and not _conf and close < _bL
        _conf := true
    if _have and _dir == 1 and not _conf and close > _bH
        _conf := true
    float _wk = (_dir == -1 and _have) ? (_tip - _bH) / math.max(ta.atr(14), 1e-10) : (_dir == 1 and _have) ? (_bL - _tip) / math.max(ta.atr(14), 1e-10) : 0.0
    float _score = 20.0 + math.min(25.0, _wk * 15.0) + (_conf ? 30.0 : 0.0) + (_wk > 1.0 ? 15.0 : 0.0) + (_wk > 1.5 ? 10.0 : 0.0)
    [_have ? _tip : na, _mid, _dir, _have ? 1 : 0, _score]

f_phys(simple int _atrL, simple int _effL, simple float _effT, simple float _dispT, simple float _convM) =>
    float _atr = ta.atr(_atrL)
    float _vel = ta.ema(close - close[1], 3)
    float _acc = _vel - _vel[1]
    float _cvx = _acc - _acc[1]
    float _csm = ta.ema(_cvx, 3)
    float _mv  = math.abs(close - close[_effL])
    float _ps  = math.sum(math.abs(close - close[1]), _effL)
    float _eff = _ps > 0 ? _mv / _ps : 0.0
    float _dsp = (high - low) / math.max(_atr, 1e-10)
    float _cth = _atr * _convM
    bool  _bImp = _eff > _effT and _vel > _vel[1] and _acc > 0 and close > open and _dsp > _dispT
    bool  _rImp = _eff > _effT and _vel < _vel[1] and _acc < 0 and close < open and _dsp > _dispT
    bool  _bDec = math.abs(_acc) < math.abs(_acc[1]) * 0.8 and _vel > 0
    bool  _rDec = math.abs(_acc) < math.abs(_acc[1]) * 0.8 and _vel < 0
    bool  _bCS  = _csm >  _cth and _csm[1] <=  _cth
    bool  _rCS  = _csm < -_cth and _csm[1] >= -_cth
    bool  _vd70 = math.abs(_vel) < math.abs(_vel[1]) * 0.7
    bool  _vd50 = math.abs(_vel) < math.abs(_vel[1]) * 0.5
    [_atr, _vel, _acc, _cvx, _csm, _eff, _dsp, _bImp, _rImp, _bDec, _rDec, _bCS, _rCS, _vd70, _vd50]

// F72 — RECURSIVE CURVE TREE  (event-generated CurveNodes — the real engine seed)
//   Recursion is EVENT-generated, not timeframe-generated: a Phase-2 CHoCH against
//   the owning node spawns a CHILD curve (same lifecycle, opposite orientation).
//   Ownership = the dominant-ENERGY node (Axiom 4). A child that keeps making
//   progress GAINS energy and owns (→ transfer); one that stalls decays and dies
//   (→ merge back into parent). FIRST-PASS: structure is solid; energy dynamics are
//   a calibratable heuristic. Additive — not yet wired to the phase labels.
// ══════════════════════════════════════════════════════════════════════════════
type CurveNode
    int    id      = 0
    int    parent  = -1
    int    dir     = 0
    float  origin  = na
    float  extreme = na
    float  energy  = 0.0
    bool   alive   = true
    int    depth   = 0
    string state   = ""      // F72 step 1: the PHASE the curve owns (emergent, Principle 1/14)
    int    bar     = 0       // bar_index at birth — anchors the curve on the chart
// ── f_nodeState — phases EMERGE from the node (Principle 1: curve → phase) ─────
//   Maps the node's own energy / lifecycle / recursion depth (+ ambient curve
//   compression & maturity) onto a Principle-14 phase label. Recursion depth > 0
//   means this node is itself a child curve, so it lives in the Transition family.
//   STEP 1: this is OBSERVED beside the legacy phase label — not yet the source of
//   truth. Calibrate against live behaviour before flipping the label's source.
f_nodeState(int _d, float _e, int _dep, float _cmp, float _mat) =>
    _dep > 0 ? (_e >= 70.0 ? "Transition · recursive expansion" : _e >= 40.0 ? "Transition · recursive induction" : "Transition · recursive liquidation") : _mat < 12.0 ? "Point 4 Origin" : (_e >= 78.0 and _mat >= 70.0) ? (_d == 1 ? "New High" : _d == -1 ? "New Low" : "Climax") : _mat < 35.0 ? "Expansion" : _mat < 55.0 ? "Expansion Pre-Convexity" : _e >= 55.0 ? "Expansion Induction" : _e >= 35.0 ? "Expansion Liquidity" : _cmp >= 60.0 ? "Retracement Pre-Convexity" : _e >= 18.0 ? "Retracement Induction" : "Retracement"
var array<CurveNode> _tree = array.new<CurveNode>()
var int _nodeSeq = 0
// pre-owner (max-energy living node) — drives child-spawn direction
int   _preOwn = -1
float _preE   = -1.0
if array.size(_tree) > 0
    for _ti = 0 to array.size(_tree) - 1
        CurveNode _tn = array.get(_tree, _ti)
        if _tn.alive and _tn.energy > _preE
            _preE := _tn.energy
            _preOwn := _ti
// seed / RE-SEED the root curve whenever no living curve owns price (a fully
// dissipated curve gives birth to a new one — Axiom 0: dissipate → transition → recur)
if _preOwn < 0 and displayWaveDir_M5 != 0 and not na(se5_inv)
    _nodeSeq += 1
    array.push(_tree, CurveNode.new(_nodeSeq, -1, displayWaveDir_M5, se5_inv, close, math.max(40.0, nz(ede_expansionEnergy, 60.0)), true, 0, bar = bar_index))
// event-generated CHILD: a Phase-2 CHoCH against the owner spawns an inverse curve
if _preOwn >= 0
    CurveNode _po = array.get(_tree, _preOwn)
    if (_po.dir == 1 and bearCHoCH) or (_po.dir == -1 and bullCHoCH)
        _nodeSeq += 1
        array.push(_tree, CurveNode.new(_nodeSeq, _po.id, -_po.dir, close, close, math.max(25.0, nz(ede_expansionEnergy, 50.0) * 0.85), true, _po.depth + 1, bar = bar_index))
// update living nodes — energy rises on progress (continuation), decays when stalled
if array.size(_tree) > 0
    for _ti = 0 to array.size(_tree) - 1
        CurveNode _tn = array.get(_tree, _ti)
        if _tn.alive
            bool _prog = _tn.dir == 1 ? high > nz(_tn.extreme, high) : low < nz(_tn.extreme, low)
            _tn.extreme := _tn.dir == 1 ? math.max(nz(_tn.extreme, high), high) : math.min(nz(_tn.extreme, low), low)
            _tn.energy  := _prog ? math.min(100.0, _tn.energy + 7.0) : math.max(0.0, _tn.energy - 2.0)
            _tn.state   := f_nodeState(_tn.dir, _tn.energy, _tn.depth, gCurve.compress, gCurve.maturity)
            if _tn.energy <= 2.0
                _tn.alive := false
while array.size(_tree) > 60
    array.shift(_tree)
// tree summary + final (emergent) owner = dominant-energy living node
int   _treeAlive = 0
int   _treeDepth = 0
int   _ownF  = -1
float _ownFE = -1.0
if array.size(_tree) > 0
    for _ti = 0 to array.size(_tree) - 1
        CurveNode _tn = array.get(_tree, _ti)
        if _tn.alive
            _treeAlive += 1
            _treeDepth := math.max(_treeDepth, _tn.depth)
            if _tn.energy > _ownFE
                _ownFE := _tn.energy
                _ownF  := _ti
int   _ownDirT   = _ownF >= 0 ? array.get(_tree, _ownF).dir    : 0
int   _ownDepthT = _ownF >= 0 ? array.get(_tree, _ownF).depth  : 0
